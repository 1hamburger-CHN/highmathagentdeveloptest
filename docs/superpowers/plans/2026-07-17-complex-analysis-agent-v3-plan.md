# backend/app/animation/templates/cr_equations.py
"""C-R Equations visualisation — directional derivative comparison."""

from app.animation.base import BaseManimTemplate


class CREquationsTemplate(BaseManimTemplate):
    template_name = "CREquations"
    concept_ids = ["complex-2.2"]
    scene_class_name = "CREquationsScene"

    def validate(self, params: dict) -> bool:
        required = {"function"}
        return required.issubset(params) and bool(params.get("function"))

    def build_scene_code(self, params: dict) -> str:
        import json
        func_val = json.dumps(params.get("function", "z**2"))
        point_val = json.dumps(params.get("point", "1+i"))
        return f'''"""C-R Equations animation — auto-generated by AnimationGenerator."""
import numpy as np
from manim import *

class {self.scene_class_name}(ThreeDScene):
    def construct(self):
        func_str = {func_val}
        point_str = {point_val}

        title = Text("Cauchy-Riemann 方程可视化", font_size=36, color=WHITE)
        title.to_edge(UP)
        self.play(Write(title), run_time=1.5)

        plane = ComplexPlane(
            x_range=[-3, 3, 1], y_range=[-3, 3, 1],
            background_line_style={{"stroke_color": BLUE_D, "stroke_width": 0.5, "stroke_opacity": 0.3}},
        ).add_coordinates()
        plane.scale(0.7)
        self.play(Create(plane), run_time=1.5)

        try:
            z0 = complex(point_str.replace("i", "j"))
        except ValueError:
            z0 = complex(1, 1)
        point_dot = Dot(plane.n2p(z0), color=YELLOW, radius=0.08)
        point_label = MathTex("z_0", font_size=28, color=YELLOW)
        point_label.next_to(point_dot, UR, buff=0.15)
        self.play(FadeIn(point_dot), Write(point_label), run_time=1)

        # Approach along real axis
        x_dots = VGroup()
        for dx in np.linspace(0.5, 0.05, 12):
            dot = Dot(plane.n2p(z0 + complex(dx, 0)), radius=0.04, color=RED)
            x_dots.add(dot)
        x_label = Text("实轴方向逼近", font_size=24, color=RED)
        x_label.to_corner(DL, buff=0.5)
        self.play(LaggedStart(*[FadeIn(d) for d in x_dots], lag_ratio=0.05), Write(x_label), run_time=2)

        # Approach along imaginary axis
        y_dots = VGroup()
        for dy in np.linspace(0.5, 0.05, 12):
            dot = Dot(plane.n2p(z0 + complex(0, dy)), radius=0.04, color=GREEN)
            y_dots.add(dot)
        y_label = Text("虚轴方向逼近", font_size=24, color=GREEN)
        y_label.next_to(x_label, RIGHT, buff=0.3)
        self.play(LaggedStart(*[FadeIn(d) for d in y_dots], lag_ratio=0.05), Write(y_label), run_time=2)

        # C-R formula
        cr_formula = MathTex(
            r"{{\\partial u \\over \\partial x}} = {{\\partial v \\over \\partial y}},\\quad " +
            r"{{\\partial u \\over \\partial y}} = -{{\\partial v \\over \\partial x}}",
            font_size=32, color=WHITE
        )
        cr_formula.to_edge(DOWN, buff=0.6)
        box = SurroundingRectangle(cr_formula, color=YELLOW, buff=0.2)
        self.play(Write(cr_formula), Create(box), run_time=2.5)

        conclusion = Text(
            "解析函数 → 导数与方向无关 → C-R 条件是必要且充分的",
            font_size=24, color=YELLOW
        )
        conclusion.next_to(cr_formula, UP, buff=0.4)
        self.play(Write(conclusion), run_time=2)
        self.wait(2)
'''

import numpy as np
from manim import *

class {self.scene_class_name}(Scene):
    def construct(self):
        func_str = {func_val}

        term_labels = VGroup()
        for n in range(max_terms):
            label = MathTex(
                f"T_{{{n+1}}}: \\frac{{z^{{{n+1}}}}}{{{n+1}}}",
                font_size=20, color=WHITE
            )
            label.scale(0.7)
            label.to_corner(UR, buff=0.3).shift(DOWN * n * 0.4)
            term_labels.add(label)

        self.play(LaggedStart(*[Write(t) for t in term_labels], lag_ratio=0.2), run_time=2)

        # Sequential approximation
        for n in range(1, min(max_terms, 6)):
            dot = Dot(plane.n2p(complex(0.5 + 1.2, 0)), color=GREEN, radius=0.06)
            n_text = Text(f"T_{n}", font_size=22, color=GREEN)
            n_text.next_to(dot, RIGHT, buff=0.3)
            self.play(FadeIn(dot), Write(n_text), run_time=0.5)
            self.remove(dot, n_text)

        conclusion = Text("逐项叠加 → 在收敛圆内逼近解析函数", font_size=24, color=YELLOW)
        conclusion.to_edge(DOWN, buff=0.6)
        self.play(Write(conclusion), run_time=1.5)
        self.wait(2)

// frontend/src/components/profile/RadarChart.tsx
"use client";

interface RadarProps {
  data: { label: string; value: number }[];
  size?: number;
}

export default function RadarChart({ data, size = 280 }: RadarProps) {
  const cx = size / 2, cy = size / 2, r = size * 0.38;
  const n = data.length;
  const angleStep = (2 * Math.PI) / n;

  const getPoint = (i: number, val: number) => {
    const angle = i * angleStep - Math.PI / 2;
    return { x: cx + r * val * Math.cos(angle), y: cy + r * val * Math.sin(angle) };
  };

  const rings = [0.25, 0.5, 0.75, 1.0].map((scale) => {
    const pts = Array.from({ length: n }, (_, i) => {
      const p = getPoint(i, scale);
      return `${p.x},${p.y}`;
    }).join(" ");
    return <polygon key={scale} points={pts} fill="none" stroke="#e5e7eb" strokeWidth="1" />;
  });

  const axes = Array.from({ length: n }, (_, i) => {
    const p = getPoint(i, 1);
    return <line key={`axis-${i}`} x1={cx} y1={cy} x2={p.x} y2={p.y} stroke="#e5e7eb" strokeWidth="1" />;
  });

  const dataPts = Array.from({ length: n }, (_, i) => {
    const p = getPoint(i, data[i].value);
    return `${p.x},${p.y}`;
  }).join(" ");

  const labels = data.map((d, i) => {
    const p = getPoint(i, 1.15);
    return (
      <text key={`label-${i}`} x={p.x} y={p.y} textAnchor="middle" dominantBaseline="middle"
            className="fill-gray-600" fontSize="11" fontFamily="system-ui">
        {d.label.length > 5 ? d.label.slice(0, 5) + "…" : d.label}
      </text>
    );
  });

  return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      {rings}{axes}
      <polygon points={dataPts} fill="rgba(59,130,246,0.2)" stroke="#3b82f6" strokeWidth="2" />
      {labels}
    </svg>
  );
}

from pathlib import Path

@router.get("/list")
async def list_animations():
    anim_dir = Path("static/animations")
    if not anim_dir.exists():
        return {"animations": []}
    animations = []
    for mp4 in sorted(anim_dir.glob("*.mp4"), key=lambda p: p.stat().st_mtime, reverse=True):
        template_name = mp4.name.split("_")[0]
        animations.append({
            "name": mp4.name,
            "url": f"/animations/{mp4.name}",
            "template": template_name,
            "size_kb": round(mp4.stat().st_size / 1024),
            "created_at": mp4.stat().st_mtime,
        })
    return {"animations": animations}
